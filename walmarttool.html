        // Load data from localStorage

        function loadData() {

            try {

                const savedProducts = localStorage.getItem('walmartProducts');

                if (savedProducts) {

                    products = JSON.parse(savedProducts);

                    updateProductsList();

                }

            } catch (e) {

                console.error('Error loading saved products:', e);

            }

        }


        // Save data to localStorage

        function saveData() {

            try {

                localStorage.setItem('walmartProducts', JSON.stringify(products));

            } catch (e) {

                console.error('Error saving products:', e);

            }

        }


        // Calculate Walmart fees

        function calculateWalmartFees(sellingPrice, category, weight, sellerType) {

            const price = parseFloat(sellingPrice) || 0;

            const shippingWeight = parseFloat(weight) || 1;

            

            const referralFees = {

                'electronics': 8.0, 'beauty': 8.0, 'pets': 8.0, 'home': 15.0,

                'clothing': 16.0, 'toys': 8.0, 'sports': 8.0, 'books': 15.0,

                'automotive': 12.0, 'health': 8.0, 'baby': 8.0, 'grocery': 8.0,

                'jewelry': 20.0, 'office': 15.0

            };


            const referralFeeRate = referralFees[category] || 15.0;

            const referralFee = (price * referralFeeRate) / 100;


            let fulfillmentFee = 0;

            if (sellerType === 'wfs') {

                if (shippingWeight <= 1) fulfillmentFee = 3.26;

                else if (shippingWeight <= 2) fulfillmentFee = 4.75;

                else if (shippingWeight <= 3) fulfillmentFee = 5.40;

                else if (shippingWeight <= 5) fulfillmentFee = 6.75;

                else fulfillmentFee = 6.75 + (shippingWeight - 5) * 0.85;

            }


            const paymentFee = (price * 0.029) + 0.30;


            return {

                referralFee, fulfillmentFee, paymentFee,

                totalFees: referralFee + fulfillmentFee + paymentFee

            };

        }


        // Calculate profit metrics

        function calculateProfitMetrics(product) {

            const selling = parseFloat(product.sellingPrice) || 0;

            const cost = parseFloat(product.costPrice) || 0;

            const competitors = parseInt(product.competitors) || 0;

            const sales = parseInt(product.estimatedSales) || 0;

            const rank = parseInt(product.walmartRank) || 999999;


            const fees = calculateWalmartFees(selling, product.category, product.shippingWeight, product.sellerType);

            const totalCosts = cost + fees.totalFees;

            const profit = selling - totalCosts;

            const profitMargin = selling > 0 ? (profit / selling) * 100 : 0;

            const roi = cost > 0 ? (profit / cost) * 100 : 0;

            const estimatedMonthlyProfit = profit * sales;


            let competitionLevel = 'Low';

            let competitionColor = 'text-green';

            if (competitors > 50) {

                competitionLevel = 'High';

                competitionColor = 'text-red';

            } else if (competitors > 20) {

                competitionLevel = 'Medium';

                competitionColor = 'text-yellow';

            }


            let profitabilityScore = 0;

            if (profitMargin > 0) profitabilityScore += 10;

            if (profitMargin > 15) profitabilityScore += 10;

            if (profitMargin > 25) profitabilityScore += 10;

            if (profitMargin > 35) profitabilityScore += 10;

            if (roi > 50) profitabilityScore += 10;

            if (roi > 100) profitabilityScore += 10;

            if (competitors < 10) profitabilityScore += 20;

            else if (competitors < 25) profitabilityScore += 15;

            else if (competitors < 50) profitabilityScore += 10;

            if (sales > 5) profitabilityScore += 5;

            if (sales > 20) profitabilityScore += 5;


            let recommendation = 'Not Recommended';

            let recommendationColor = 'text-red';

            let recommendationIcon = '❌';


            if (profitabilityScore >= 75) {

                recommendation = 'Highly Recommended';

                recommendationColor = 'text-green';

                recommendationIcon = '✅';

            } else if (profitabilityScore >= 55) {

                recommendation = 'Worth Considering';

                recommendationColor = 'text-yellow';

                recommendationIcon = '⚠️';

            }


            return {

                profit, profitMargin, roi, estimatedMonthlyProfit,

                competitionLevel, competitionColor, profitabilityScore,

                recommendation, recommendationColor, recommendationIcon, fees

            };

        }

        // Remove product

        function removeProduct(id) {

            if (confirm('Are you sure you want to remove this product?')) {

                products = products.filter(p => p.id !== id);

                saveData();

                updateProductsList();

                

                // Hide analysis if current product was deleted

                if (currentAnalysis && currentAnalysis.product.id === id) {

                    document.getElementById('analysisSection').classList.add('hidden');

                    currentAnalysis = null;

                }

            }

        }


        // Analyze product

        function analyzeProduct(id) {

            const product = products.find(p => p.id === id);

            if (!product) return;


            const metrics = calculateProfitMetrics(product);

            currentAnalysis = { product, metrics };

            

            document.getElementById('analysisTitle').textContent = `Walmart Analysis: ${product.name}`;

            document.getElementById('analysisContent').innerHTML = generateAnalysisHTML(product, metrics);

            document.getElementById('analysisSection').classList.remove('hidden');

            

            // Scroll to analysis

            document.getElementById('analysisSection').scrollIntoView({ behavior: 'smooth' });

        }
